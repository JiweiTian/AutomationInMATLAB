function out = fun_e(p,j)
    % wartości N, a, b, na, nb muszą już być w workspace'ie
    global N a b na nb nu ny
    if(j==1 && nb==1)
        out = 0;
    elseif(j==1 && nb>1)
        out = fun_g(p,j);
    elseif(j>=2 && j<=(nb-1) && j<nb && nb>1)
        out = fun_g(p,j) - fun_g(p,j-1);
    elseif(j==nb && nb>1)
        out = -fun_g(p,j-1);
    else
        error('Coś poszło nie tak! Parametry wywołania:\np=%d; j=%d; nb=%d;\n',p,j,nb);
    end
end

function out = fun_f(p,j,m)
    % wartości N, a, b, na, nb muszą już być w workspace'ie
    disp(N)
    global N a b na nb
    disp(N)
    if(p == 1)
        if(j==0)
            out = 1 - a(m,1);
        elseif(j>=1 && j<=(na-1))
            out = a(m,j)-a(m,j+1);
        elseif(j==na)
            out = a(m,j);
        else
            error('Coś poszło nie tak1! Parametry wywołania:\np=%d; j=%d; na=%d;\n',p,j,na);
        end
    elseif(p>=2 && p<=N)
        if(j>=0 && j<=(na-1))
            out = fun_f(p-1,0,m)*fun_f(1,j,m)+fun_f(p-1,j+1,m);
        elseif(j==na)
            out = fun_f(p-1,0,m)*fun_f(1,j,m);
        else
            error('Coś poszło nie tak2! Parametry wywołania:\np=%d; j=%d; na=%d;\n',p,j,na);
        end
    else
        error('Coś poszło nie tak3! Parametry wywołania:\np=%d; j=%d; N=%d;\n',p,j,N);
    end
end

function out = fun_g(p,j)
    % wartości N, a, b, na, nb muszą już być w workspace'ie
    global N a b na nb nu ny
    if(p == 1)
        if(j>=0 && j<=(nb-1))
            out = b(j+1);
        else
            error('Coś poszło nie tak1! Parametry wywołania:\np=%d; j=%d; nb=%d;\n',p,j,nb);
        end
    elseif(p>=2 && p<=N)
        if(j>=(1-p) && j<=(-1))
            out = fun_g(p-1,j+1);
        elseif(j>=0 && j<=(nb-2))
            out = fun_f(p-1,0)*fun_g(1,j)+fun_g(p-1,j+1);
        elseif(j==(nb-1))
            out = fun_f(p-1,0)*fun_g(1,j);
        else
            error('Coś poszło nie tak2! Parametry wywołania:\np=%d; j=%d; nb=%d;\n',p,j,nb);
        end
    else
        error('Coś poszło nie tak3! Parametry wywołania:\np=%d; j=%d; nb=%d;\n',p,j,nb);
    end
end
